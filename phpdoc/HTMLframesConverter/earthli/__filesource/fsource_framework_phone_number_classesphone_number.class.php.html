  21. /**
  22.  * Phone Number representation class
  23.  * 
  24.  * This class holds the number input and its alternative representation. The following
  25.  * representations should be available:
  26.  *     international number,
  27.  *     input number,
  28.  *     international normalized number,
  29.  *     normalized number.
  30.  * The information will only be requested from the databases when they are requested the
  31.  * first time from the class.
  32.  * 
  33.  * 0030            2310            205 385
  34.  * COUNTRY CODE    REGION CODE     SUBSCRIBER NUMBER
  35.  * 
  36.  * COUNTRY CODE:
  37.  * The country code is the part of an international number that identifies the country.
  38.  * REGION CODE:
  39.  * The region code is the part of the number that identifies the geographical location or the network where the subscriber number is located.
  40.  * SUBSCRIBER NUMBER:
  41.  * The subscriber number is the part of the phone number that identifies the subscriber. It does not include the region code.
  42.  * 
  43.  * EXIT DIGITS:
  44.  * The exit digits are country specific digits that are required to place an international call. (The international representation of the exit digit is the leading '+')
  45.  * 
  46.  * TRUNK CODE:
  47.  * The trunk code is the part of the number that defines if the following digits are part of the region code or part of the subscriber number.
  48.  * 
  49.  * LOCAL NUMBER:
  50.  * A local number is the phone number that contains the phone number that is used to call within a country. It must not contain the country dial code.
  51.  * It will contain any region code and if applicable (depending on the country) the trunk code.
  52.  * 
  53.  * INTERNATIONAL NUMBER:
  54.  * A international number is the representation of a number including the region code, country code and the country specific exit-digits (the alternative to '+").
  55.  * This numbers are international but can not be dialed from all countries.
  56.  * 
  57.  * INTERNATIONAL NORMALIZED NUMBER:
  58.  * A international normalized number is the representation of the phone number that allows it to call across countries. It must contain the country code and the leading "+".
  59.  * 
  60.  * NORMALIZED NUMBER:
  61.  * A normalized number is the international number including the country dial code but without the leading exit digit or "+".
  62.  * 
  63.  * 
  64.  * Error codes:
  65.  * 101 ... "country_3_letter was not set"
  66.  * 102 ... "country_3_letter was not string"
  67.  * 103 ... "parameter country_3_letter was not a 3 letter string"
  68.  * 104 ... "input_number was not set"
  69.  * 105 ... "input_number was not string"
  70.  * 150 ... "sql object not specified in costructor"
  71.  * 201 ... "country_3_letter was set to NULL"
  72.  * 202 ... "input_number was NOT set"
  73.  * 301 ... "Fetching data failed. Internal SQL error ".$errno
  74.  * 401 ... "missing parameters, cannot continue to process number"
  75.  * 402 ... "false exit dialcode array, cannot continue to process number"
  76.  * 403 ... "tried to compare two non-phone number objects"
  77. **/

  450.         return TRUE;

  750.     
  751.     
  752. /**    
  753.      * This method processes the input_number and stores in valid format.
  754.      * 
  755.      * This method accesses the following tables:
  756.      *  Country_Exit_Dialcode
  757.      *  Country_Dialcodes
  758.      *  Country_Trunk_Code
  759.      * to fetch the required information for the country specified in the
  760.      * parameter $this->_country_3_letter. If the correct fields could be retrieved
  761.      * then we can continue processing the validated input number and normalize it.
  762.      * 
  763.      * @access private
  764.      * 
  765.      * @param  void 
  766.      * @return boolean 
  767.     ***/
  768.     private function _fetch_dialcodes()
  769.     {
  770.         // step 1: make sure we have the country_3_letter code AND the validated version of the number
  771.         $this->_country_3_letter = $this->get_normalized_country();
  772.         
  773.         if(isset($this->_country_3_letter) === TRUE /*&& isset($this->_validated_number) === TRUE*/ && (isset($this->_country_code) === FALSE))
  774.         {
  775.             // step 2: fetch from the database the exit_dialcode, international_dialcode, extended_dialcode, trunk_dialcode for this specific country
  776.             parent::debug2("trying to fetch data from mysql");
  777.             
  778.             // just in case someone forgot to pass us a database name we use a default value here
  779.             $db_name = '';
  780.             if(isset($this->_sql_db) === TRUE && is_null($this->_sql_db) === FALSE)
  781.             {
  782.                 $db_name = $this->_sql_db . '.';
  783.             }
  784.             // get the country_3_letter, country exit_dialcode, country international_dialcode, country extended_dialcode
  785.             // trunk dialcode for a specific country to normalize the received number into international format before
  786.             // further processing is possible
  787.             $query = "SELECT ".$db_name."Country_Exit_Dialcode.country_3_letter, \n"
  788.                     ."       ".$db_name."Country_Exit_Dialcode.exit_dialcode, \n"
  789.                     ."       ".$db_name."Country_Dialcodes.international_dialcode, \n"
  790.                     ."       ".$db_name."Country_Dialcodes.extended_dialcode, \n"
  791.                     ."       ".$db_name."Country_Trunk_Code.trunk_dialcode \n"
  792.                     ."FROM   ".$db_name."Country_Exit_Dialcode, \n"
  793.                     ."       ".$db_name."Country_Dialcodes, \n"
  794.                     ."       ".$db_name."Country_Trunk_Code \n"
  795.                     ."WHERE  ".$db_name."Country_Exit_Dialcode.country_3_letter = '".$this->_country_3_letter."' AND \n"
  796.                     ."       ".$db_name."Country_Dialcodes.country_3_letter = '".$this->_country_3_letter."' AND \n"
  797.                     ."       ".$db_name."Country_Trunk_Code.country_3_letter = '".$this->_country_3_letter."'";
  798.             
  799.             // send query to db and get result
  800.             $errno = NULL;
  801.             $errtext = NULL;
  802.             
  803.             $query_result = $this->_sql_obj->get_query_result($query, $errno, $errtext);
  804.             
  805.             if($errno !== NULL)
  806.             {
  807.                 parent::report_error(301, "Fetching data failed. Internal SQL error ".$errno.': '.$errtext);
  808.                 $this->_error = TRUE;
  809.             }
  810.             
  811.             if($query_result['count'] > 0)
  812.             {
  813.                 parent::debug2("Found CLI Country information: ". print_r($query_result['data'], TRUE));
  814.                 parent::debug2("trunk code = '".$query_result['data'][0]['trunk_dialcode']."'");
  815.                 parent::debug2("trunk type= '".gettype($query_result['data'][0]['trunk_dialcode'])."'");
  816.                 
  817.                 $this->_country_code = $query_result['data'][0]['international_dialcode'];
  818.                 $this->_exit_dialcode = array();
  819.                 $this->_trunk_code = array();
  820.                 
  821.                 for($i = 0; $i < $query_result['count']; $i++)
  822.                 {
  823.                     if(in_array($query_result['data'][$i]['exit_dialcode'], $this->_exit_dialcode) === FALSE)
  824.                     {
  825.                         $this->_exit_dialcode[] = $query_result['data'][$i]['exit_dialcode'];
  826.                     }
  827.                     if(in_array($query_result['data'][$i]['trunk_dialcode'], $this->_trunk_code) === FALSE)
  828.                     {
  829.                         $this->_trunk_code[] = $query_result['data'][$i]['trunk_dialcode'];
  830.                     }
  831.                 }
  832.                 // sort array (order is important for normalization)
  833.                 usort($this->_trunk_code, array($this, '_sort_dialcode'));
  834.             }
  835.             else
  836.             {
  837.                 parent::debug2("no exit dialcode / country code information found in DB, cannot normalize number, EXIT NOW !!!");
  838.             }
  839.             
  840.             return TRUE;
  841.         }
  842.         else
  843.         {
  844.             parent::debug2("could not find country code in countries table! !!!");
  845.             return FALSE;
  846.         }
  847.     }

  904.     
  905.     
  906. /**    
  907.      * Convert an input number to a valida input number format
  908.      * 
  909.      * This method is used to convert an unprocessed input number to a validated number format.
  910.      * We have two separate methods. The first one uses the _normalize_number method and the second
  911.      * one is simply using a replacement table (is work in progress).
  912.      * 
  913.      * @access private
  914.      * 
  915.      * @param void 
  916.      * @return string 
  917.     ***/
  918.     private function _input_to_valid($number, $method = 0)
  919.     {
  920.         parent::debug2("called with parameters: number = [".$number."] method=".$method."]");
  921.         
  922.         if(isset($method) === FALSE)
  923.         {
  924.             $method = 0;
  925.         }
  926.         
  927.         switch ($method)
  928.         {
  929.             case 0:
  930.                 $success_fetching = $this->_fetch_dialcodes();
  931.                 if($success_fetching)
  932.                 {
