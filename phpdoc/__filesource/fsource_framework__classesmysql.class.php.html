  303.     public function connection(&$errno, &$errstr)
  304.     {
  305.         parent::debug2("Called method ".__FUNCTION__);
  306.         if($this->dbg_intern_main > 0) echo "<pre>(Line: ".__LINE__.") ".get_class($this)."->".__FUNCTION__." - Called method ".__FUNCTION__.".</pre>\n";

  308.         // check for a persistant class error

  310.         {
  311.             $errno = $this->errno;
  312.             $errstr = $this->errtext;

  314.             return FALSE;
  315.         }
  316.         
  317.         parent::debug("Connect to db_type=".$this->db_type." db_host=".$this->db_host." db_user=".$this->db_user);
  318.         $errno = $errstr = NULL;
  319.         
  320.         
  321.         // Check if SQL is connect or not
  322.         if(empty($this->sql_handle) === TRUE)
  323.         {
  324.             // Define SQL Variables
  325.             if(isset($this->db_host) === FALSE || isset($this->db_name) === FALSE || isset($this->db_user) === FALSE || isset($this->db_pass) === FALSE)
  326.             {
  327.                 $errno = 100;
  328.                 $errstr = "Missing connection informations (server, db, user or password)";
  329.                 parent::error("Connect ... Missing connection informations");
  330.                 return FALSE;
  331.             }
  332.             // SQL Connection
  333.             parent::performance("", "start");
  334.             if ($this->db_persistent === TRUE)
  335.             {
  336.                 $this->sql_handle = @mysql_pconnect($this->db_host,$this->db_user,$this->db_pass);
  337.             }
  338.             else
  339.             {
  340.                 $this->sql_handle = @mysql_connect($this->db_host,$this->db_user,$this->db_pass, TRUE);
  341.             }
  342.             parent::performance(get_class($this)." - Connect to ... ".$this->db_host, "stop");
  343.             
  344.             // check for the sql connection
  345.             if(empty($this->sql_handle) === TRUE)
  346.             {
  347.                 $errno = 101;
  348.                 $errstr = "Connect to mysql Server Error: #".mysql_errno().": ".mysql_error();
  349.                 // not required
  350.                 //
  351.                 parent::error($errstr);
  352.                 return FALSE;
  353.             }
  354.             
  355.             // Select Database
  356.             $this->db_link = @mysql_select_db($this->db_name, $this->sql_handle);
  357.             if(!$this->db_link)
  358.             {
  359.                 $errno = 102;
  360.                 
  361.                 
  362.                 $errstr = "Connect to DB Error. DBName = $this->db_name. #".mysql_errno().": ".mysql_error();
  363.                 parent::error($errstr);
  364.                 return FALSE;
  365.             }
  366.  
  367.             if($this->sucess_debug == 1)
  368.             {
  369.                 $errno = 902;
  370.                 $errstr = "Connect to mysql DB Sucess. DBName = $this->db_name";
  371.             }
  372.             
  373.             // set internal character for queries and results and check if OK

  377.             {
  378.                 if(function_exists('mysql_set_charset') === TRUE)
  379.                 {

  381.                 }
  382.             }
  383.             else

  385.                 // @see http://dev.mysql.com/doc/refman/5.0/en/charset-connection.html

  387.                 // and collation_connection to character_set_database and collation_database.
  388.                 $this->query("SET CHARACTER SET $this->charset_db", $errno, $errstr);
  389.                 if($errno !== NULL)
  390.                 {
  391.                     $ok = FALSE;
  392.                 }
  393.                 // SET NAMES indicates what character set the client will use to send SQL statements to the server.
  394.                 $this->query("SET NAMES '$this->charset_db'", $errno, $errstr);
  395.                 if($errno !== NULL)
  396.                 {
  397.                     $ok = FALSE;
  398.                 }
  399.             }
  400.             
  401.             if($ok === FALSE)
  402.             {
  403.                 // TODO: should be moved to warn() when it becomes available
  404.                 parent::debug("WARNING: character set could not be set on the connection");
  405.             }
  406.             //
  407.             
  408.             parent::debug2("Connect ... SUCESS");
  409.             return TRUE;
  410.         }
  411.         else
  412.         {
  413.             // already connected, return TRUE
  414.             return TRUE;
  415.         }
  416.     }
